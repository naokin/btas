377 template<typename T, size_t N>
378 TVector<T, N> transpose(const TVector<T, N>& vec, int K) {
379  assert(K >= 0 && K <= N);
380  TVector<T, N> vtr;
381  for(int i = 0; i < N-K; ++i) vtr[i] = vec[i+K];
382  for(int i = N-K; i < N; ++i) vtr[i] = vec[i+K-N];
383  return std::move(vtr);
384 }
